import random
import string

size = 1_000_000  # target size in characters
text = ''.join(random.choices(string.ascii_letters + string.digits + ' ', k=size))

# Save to file so the user can access it
file_path = "/mnt/data/random_text_1mb.txt"
with open(file_path, "w", encoding="utf-8") as f:
    f.write(text)
